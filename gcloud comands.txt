gcloud compute networks create customvpn \
--subnet-mode=custom \
--bgp-routing-mode=regional\

gcloud compute networks subnets create private-subnet  \
--network=customvpn \
--range=10.128.0.0/20 \
--enable-flow-logs \
--enable-private-ip-google-access \
--purpose=PRIVATE \
--region=us-west1 \
--secondary-range range2=192.168.64.0/24 \

gcloud compute instances create private-instance \
--boot-disk-auto-delete \
--machine-type=n1-standard-1 \
--metadata startup-script= \
"sudo apt-get install tinyproxy && sudo service tinyproxy restart" \
--network=customvpn \
--subnet=private-subnet \
--preemptible \
--tags tag-firewall \
--zone=us-west1-a 

gcloud compute instances delete private-instance --zone=us-west1-a
